36  class Kernels

43  inline T gaussian_cdf(T x, T mu, T sigma) const
44  {
45  // 0.5*(1.0 + std::erf((x - mu)/(sigma*std::sqrt(2.0))))
46  // Abramowitz Stegun normal CDF
47  T t = 1.0/(1.0 + 0.2316419*std::abs(x - mu)/sigma);
48  T y = t*(0.319381530 + t*(-0.356563782 + t*(1.781477937 + t*(-1.821255978 + t*1.330274429))));
49  if(x >= mu)
50  {
51  return 1.0 - gaussian_pdf(x, mu, sigma)*y*sigma;
52  }
53  else
54  {
55  return gaussian_pdf(x, mu, sigma)*y*sigma;
56  }
57  }
58  inline T epanechnikov_cdf(T x, T mu, T sigma) const
59  {
60  T z = (x - mu)/sigma;
61  if(z < -1.0)
62  return 0.0;
63  if(z > 1.0)
64  return 1.0;
65  return 0.25*(2.0 + 3.0*z - std::pow(z, 3.0));
66  }
67  inline T laplacian_cdf(T x, T mu, T sigma, T lambda) const
68  {
69  T z = (x - mu)/sigma;
70  if(z < 0.0)
71  return 0.5*std::exp(lambda*z);
72  return 1.0 - 0.5 *std::exp(-lambda*z);
73  }
74  inline T biweight_cdf(T x, T mu, T sigma) const
75  {
76  T z = (x - mu)/sigma;
77  if(z < -1.0)
78  return 0.0;
79  if(z > 1.0)
80  return 1.0;
81  return 0.9375*(z - 2.0*std::pow(z, 3.0)/3.0 + 0.2*std::pow(z, 5.0)) + 0.5;
82  }
83  inline T triweight_cdf(T x, T mu, T sigma) const
84  {
85  T z = (x - mu)/sigma;
86  if(z < -1.0)
87  return 0.0;
88  if(z > 1.0)
89  return 1.0;
90  return 1.09375*(z - std::pow(z, 3.0) + 0.6*std::pow(z, 5.0) - std::pow(z, 7.0)/7.0) + 0.5;
91  }
92  inline T uniform_cdf(T x, T mu, T sigma) const
93  {
94  if(x < mu - 0.5*sigma)
95  return 0.0;
96  if(x > mu + 0.5*sigma)
97  return 1.0;
98  return (x-mu)/sigma + 0.5;
99  }
100  inline T gaussian_pdf(T x, T mu, T sigma) const
101  {
102  return std::exp(-0.5*std::pow((x - mu)/sigma, 2.0))/(sigma*std::sqrt(2.0*pi));
103 
104 // return fmath::exp(-0.5*std::pow((x - mu)/sigma, 2.0))/(sigma*std::sqrt(2.0*pi));
105  }
106  inline T laplacian_pdf(T x, T mu, T sigma, T lambda) const
107  {
108  T z = (x - mu)/sigma;
109  return 0.5*lambda*std::exp(-std::abs(lambda*z));
110  }
111  inline T epanechnikov_pdf(T x, T mu, T sigma) const
112  {
113  T z = (x - mu)/sigma;
114  return std::abs(z) > 1.0 ? 0.0 : 0.75*(1.0-std::pow(z, 2.0))/sigma;
115  }
116  inline T biweight_pdf(T x, T mu, T sigma) const
117  {
118  T z = (x - mu)/sigma;
119  return std::abs(z) > 1.0 ? 0.0 : 0.9375*std::pow(1.0 - std::pow(z, 2.0), 2.0);
120  }
121  inline T triweight_pdf(T x, T mu, T sigma) const
122  {
123  T z = (x - mu)/sigma;
124  return std::abs(z) > 1.0 ? 0.0 : 1.09375*std::pow(1.0 - std::pow(z, 2.0), 3.0);
125  }
126  inline T uniform_pdf(T x, T mu, T sigma) const
127  {
128  return (x < mu - 0.5*sigma || x > mu + 0.5*sigma) ? 0.0 : 1.0/sigma;
129  }
130  public:
131  inline T compute_pdf(size_t kt, T x, T mu, T sigma, T lambda = 1.0) const
132  {
133  switch(kt)
134  {
135  case 1:
136  return epanechnikov_pdf(x, mu, sigma);
137  case 2:
138  return uniform_pdf(x, mu, sigma);
139  case 3:
140  return biweight_pdf(x, mu, sigma);
141  case 4:
142  return triweight_pdf(x, mu, sigma);
143  case 5:
144  return laplacian_pdf(x, mu, sigma, lambda);
145  default:
146  return gaussian_pdf(x, mu, sigma);
147  }
148  }
149  inline T compute_cdf(size_t kt, T x, T mu, T sigma, T lambda = 1.0) const
150  {
151  switch(kt)
152  {
153  case 1:
154  return epanechnikov_cdf(x, mu, sigma);
155  case 2:
156  return uniform_cdf(x, mu, sigma);
157  case 3:
158  return biweight_cdf(x, mu, sigma);
159  case 4:
160  return triweight_cdf(x, mu, sigma);
161  case 5:
162  return laplacian_cdf(x, mu, sigma, lambda);
163  default:
164  return gaussian_cdf(x, mu, sigma);
165  }
166  }

351  T parallel_pdf(InputIt first, InputIt last, const std::vector<T> &x, T lambda) const
352  {
353  T res = 0.0;
354  const auto size = last - first;
355  const auto nthreads = std::thread::hardware_concurrency();
356  const auto size_per_thread = size / nthreads;
357 
358  std::vector<std::future<T>> futures;
359  for(unsigned int i = 0; i < nthreads - 1; i++)
360  {
361  futures.emplace_back(std::async([start = first + i * size_per_thread, size_per_thread, x, lambda, this]()
362  {
363  return this->pdf_it(start, start + size_per_thread, x, lambda);
364  }));
365  }
366  futures.emplace_back(
367  std::async([start = first + (nthreads - 1) * size_per_thread, last, x, lambda, this]()
368  {
369  return this->pdf_it(start, last, x, lambda);
370  }));
371 
372  for(auto &&future : futures)
373  {
374  if(future.valid())
375  {
376  res += future.get();
377  }
378  else
379  {
380  throw std::runtime_error("Something going wrong.");
381  }
382  }
