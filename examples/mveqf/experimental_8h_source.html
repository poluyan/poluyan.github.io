85  for(size_t i = 0; i != dimension - 1; i++)

438  std::pair<size_t, TFloat> ImplicitGraphQuantile<TIndex, TFloat>::quantile_transform(const std::vector<TIndex> &layer, const std::vector<size_t> &psum, size_t ind, TFloat val01) const
439  {
440  size_t m = 0, count = grid_number[ind], step, c1 = 0, c2 = 0;
441  TFloat f1 = 0.0, f2 = 0.0, sample_size_u = static_cast<TFloat>(layer.size());
442  //auto first = grids[ind].begin();
443  //auto it = grids[ind].begin();
444  size_t it = 0, first = 0;
445 
446  while(count > 0)
447  {
448  it = first;
449  step = count / 2;
450  it += step;
451  m = it;
452  //std::advance(it, step);
453  //m = std::distance(grids[ind].begin(), it);
454 
455  c1 = psum[count_less_binary(layer, m)];
456  f1 = c1/sample_size_u;
457 
458  if(f1 < val01)
459  {
460  c2 = psum[count_less_binary(layer, m + 1)];
461  f2 = c2/sample_size_u;
462 
463  if(val01 < f2)
464  break;
465 
466  first = ++it;
467  count -= step + 1;
468  }
469  else
470  {
471  count = step;
472  }
473  }
474 
475  if(count == 0)
476  {
477  c2 = psum[count_less_binary(layer, m + 1)];
478  f2 = c2/sample_size_u;
479  }
480 
481  if(c1 == c2)
482  {
483  if(c1 == 0)
484  {
485  return std::make_pair(0, get_grid_value(ind, layer.front()) + 2.0*val01*dx[ind]);
486  }
487  if(c1 == layer.size())
488  {
489  return std::make_pair(layer.size() - 1, get_grid_value(ind, layer.back()) + 2.0*val01*dx[ind]);
490  }
491 
492  TIndex target = m;
493  auto pos = std::lower_bound(layer.begin(), layer.end(), target);
494  size_t index = std::distance(layer.begin(), pos);
495 
496  if(index > 0)
497  {
498  int curr1 = std::abs(static_cast<int>(layer[index]) - static_cast<int>(m));
499  int curr2 = std::abs(static_cast<int>(layer[index - 1]) - static_cast<int>(m));
500 
501  if(curr1 < curr2)
502  {
503  return std::make_pair(index, get_grid_value(ind, layer[index]) + 2.0*val01*dx[ind]);
504  }
505  else if(curr1 == curr2)
506  {
507  if(layer[index - 1] < layer[index])
508  return std::make_pair(index - 1, get_grid_value(ind, layer[index - 1]) + 2.0*val01*dx[ind]);
509  else
510  return std::make_pair(index, get_grid_value(ind, layer[index]) + 2.0*val01*dx[ind]);
511  }
512  else
513  {
514  return std::make_pair(index - 1, get_grid_value(ind, layer[index - 1]) + 2.0*val01*dx[ind]);
515  }
516 
517  }
518  return std::make_pair(index, get_grid_value(ind, layer[index]) + 2.0*val01*dx[ind]);
519  }
520  TIndex target = m;
521  auto pos = std::lower_bound(layer.begin(), layer.end(), target);
522  TIndex index = std::distance(layer.begin(), pos);
523  return std::make_pair(index, get_grid_value(ind, m) + (val01 - f1) * (get_grid_value(ind, m + 1) - get_grid_value(ind, m)) / (f2 - f1));
524  }
