118  bool Node<TIndex>::same_path(Node<TIndex> *p) const
119  {
120  if(this->children.size() == p->children.size())
121  {
122  for(const auto &i : this->children)
123  {
124  TIndex label = i.first;
125  if(p->children.size() == 1 && p->children.front().first != label)
126  return false;

128  [&label](const std::pair<TIndex, std::shared_ptr<Node<TIndex>>> &obj)
129  {
130  return obj.first == label;

132  if(it == p->children.end())
133  return false;

135  {
136  return false;
137  }
138  }
139  }

259  template <typename TIndex>
260  void MFSA<TIndex>::replace_or_register(const std::shared_ptr<Node<TIndex>> &p, const std::vector<TIndex> &key)
261  {
262  TIndex label = key.front();
263  auto t = p->transition_shared(label);
264  std::vector<TIndex> subvect(key.begin() + 1, key.begin() + key.size());
265  if(t->children.size() > 0 && subvect.size() > 0)
266  replace_or_register(t, subvect);
267 
268  std::shared_ptr<Node<TIndex>> en = nullptr;
269  for(size_t j = 0; j != eq.size(); j++)
270  {
271  if(eq[j].first->is_equal(t.get()))
272  {
273  en = eq[j].second;
274  break;
275  }
276  }
277  if(en == nullptr)
278  {
279  eq.push_back(std::make_pair(t, t));
280  }
281  else if(en != t)
282  {
283  for(const auto &i : t->children)
284  i.second->in_count--;
285 
286  t->in_count--;
287  en->in_count++;
288  auto it = std::find_if(p->children.begin(), p->children.end(),
289  [&label](const std::pair<TIndex, std::shared_ptr<Node<TIndex>>> &obj)
290  {
291  return obj.first == label;
292  });
293  if(it != p->children.end())
294  p->children[std::distance(p->children.begin(), it)].second = en;
295  else
296  {
297  p->children.emplace_back(std::make_pair(label, en));
298  p->children.shrink_to_fit();

338  {

354  template <typename TIndex>
355  void MFSA<TIndex>::clone_path(Node<TIndex> *pivot, const std::vector<TIndex> &to_pivot, const std::vector<TIndex> &key) const
356  {
357  Node<TIndex> *last_target = pivot->transition(key);
358  std::shared_ptr<Node<TIndex>> last = nullptr;
359  for(int i = key.size(); i >= 0; i--)
360  {
361  std::vector<TIndex> current_key;
362  if(i > 0)
363  std::copy_n(key.begin(), i, std::back_inserter(current_key));
364 
365  Node<TIndex> *target = i > 0 ? pivot->transition(current_key) : pivot;
366  std::shared_ptr<Node<TIndex>> cloned;
367 
368  if(i == 0)
369  {
370  std::vector<TIndex> key_to_pivot(to_pivot.begin(), to_pivot.begin() + to_pivot.size() - 1);
371  TIndex parent_label = to_pivot.back();
372 
373  Node<TIndex> *parent = root->transition(key_to_pivot);
374  cloned = std::make_shared<Node<TIndex>>(pivot);
375  pivot->in_count--;
376  cloned->in_count++;
377  auto it = std::find_if(parent->children.begin(), parent->children.end(),
378  [&parent_label](const std::pair<TIndex, std::shared_ptr<Node<TIndex>>> &obj)
379  {
380  return obj.first == parent_label;
381  });
382  if(it != parent->children.end())
383  parent->children[std::distance(parent->children.begin(), it)].second = cloned;
384  else
385  {
386  parent->children.emplace_back(std::make_pair(parent_label, cloned));
387  parent->children.shrink_to_fit();
388  }
389  }
390  else
391  {
392  cloned = std::make_shared<Node<TIndex>>(target);
393  }
394 
395  if(last != nullptr)
396  {
397  last_target->in_count--;
398  last->in_count++;
399  TIndex last_label = key[i]; // last = nullptr at first iteration
400  auto it = std::find_if(cloned->children.begin(), cloned->children.end(),
401  [&last_label](const std::pair<TIndex, std::shared_ptr<Node<TIndex>>> &obj)
402  {
403  return obj.first == last_label;
404  });
405 
406  if(it != cloned->children.end())
407  cloned->children[std::distance(cloned->children.begin(), it)].second = last;
408  else
409  {
410  cloned->children.emplace_back(std::make_pair(last_label, last));
411  cloned->children.shrink_to_fit();
412  }
413  last_target = target;
414  }
415  last = cloned;
416  }

419  template <typename TIndex>
420  void MFSA<TIndex>::add(const std::vector<TIndex> &key)
421  {
422  std::vector<TIndex> prefix = longest_prefix(key);
423  std::vector<TIndex> suffix(key.begin() + prefix.size(), key.begin() + key.size());
424 
425  size_t i = 0;
426  Node<TIndex> *current = root.get();
427  for(; i < prefix.size(); i++)
428  {
429  auto t = prefix[i];
430  auto it = std::find_if(current->children.begin(), current->children.end(),
431  [&t](const std::pair<TIndex, std::shared_ptr<Node<TIndex>>> &obj)
432  {
433  return obj.first == t;
434  });
435  current = it != current->children.end() ? current->transition(prefix[i]) : nullptr;
436  if(current == nullptr || current->in_count > 1)
437  {
438  break;
439  }
440  }
441 
442  if(current == root.get() || i == prefix.size())
443  {
444  remove_path(prefix);
445  }
446  else
447  {
448  std::vector<TIndex> subvect(prefix.begin(), prefix.begin() + i);
449  remove_path(subvect);
450  if(current != nullptr)
451  {
452  std::vector<TIndex> key_to_first(prefix.begin(), prefix.begin() + i + 1);
453  std::vector<TIndex> duplicate(prefix.begin() + i + 1, prefix.begin() + prefix.size());
454  clone_path(current, key_to_first, duplicate);
455  }
456  }
457  add_path(root->transition(prefix), suffix);
458  }
