184  std::pair<size_t, TFloat> ImplicitQuantile<TIndex, TFloat>::quantile_transform(trie_based::NodeCount<TIndex> *layer, size_t ind, TFloat val01) const
185  {
186  size_t m = 0, count = grid_number[ind], step, a = 0, b = 0;
187  TFloat x = 0.0, y = 0.0, p = static_cast<TFloat>(layer->count);
188  //auto first = grids[ind].begin();
189  //auto it = grids[ind].begin();
190  size_t it = 0, first = 0;
191 
192  while(count > 0)
193  {
194  it = first;
195  step = count / 2;
196  it += step;
197  m = it;
198  //std::advance(it, step);
199  //m = std::distance(grids[ind].begin(), it);
200 
201  std::tie(a, b) = count_less(layer, m);
202  x = static_cast<TFloat>(a)/p;
203 
204  if(x < val01)
205  {
206  y = static_cast<TFloat>(b)/p;
207  if(val01 < y)
208  break;
209 
210  first = ++it;
211  count -= step + 1;
212  }
213  else
214  count = step;
215  }
216  if(count == 0)
217  {
218  y = static_cast<TFloat>(b)/p;
219  }
220  if(a == b)
221  {
222  if(a == 0)
223  {
224  auto min_val_it = std::min_element(layer->children.begin(), layer->children.end(),
225  [](const std::shared_ptr<trie_based::NodeCount<TIndex>> &l,
226  const std::shared_ptr<trie_based::NodeCount<TIndex>> &r)
227  {
228  return l->index < r->index;
229  });
230  size_t min_ind = std::distance(layer->children.begin(), min_val_it);
231  //return std::make_pair(min_ind, grids[ind][layer->children[min_ind]->index] + 2.0*val01*dx[ind]);
232  return std::make_pair(min_ind, get_grid_value(ind, layer->children[min_ind]->index) + 2.0*val01*dx[ind]);
233  }
234  if(a == layer->count)
235  {
236  auto max_val_it = std::max_element(layer->children.begin(), layer->children.end(),
237  [](const std::shared_ptr<trie_based::NodeCount<TIndex>> &l,
238  const std::shared_ptr<trie_based::NodeCount<TIndex>> &r)
239  {
240  return l->index < r->index;
241  });
242  size_t max_ind = std::distance(layer->children.begin(), max_val_it);
243  //return std::make_pair(max_ind, grids[ind][layer->children[max_ind]->index] + 2.0*val01*dx[ind]);
244  return std::make_pair(max_ind, get_grid_value(ind, layer->children[max_ind]->index) + 2.0*val01*dx[ind]);
245  }
246  int diff = std::numeric_limits<int>::max();
247  size_t index = 0;
248  int min_ind = static_cast<int>(layer->children[index]->index);
249  for(size_t i = 1; i != layer->children.size(); ++i)
250  {
251  int t = static_cast<int>(layer->children[i]->index);
252  int curr = std::abs(t - static_cast<int>(m));
253  if(diff > curr)
254  {
255  diff = curr;
256  index = i;
257  min_ind = t;
258  }
259  else if(diff == curr)
260  {
261  if(min_ind > t)
262  {
263  min_ind = t;
264  index = i;
265  }
266  }
267  }
268  //return std::make_pair(index, grids[ind][layer->children[index]->index] + 2.0*val01*dx[ind]);
269  return std::make_pair(index, get_grid_value(ind, layer->children[index]->index) + 2.0*val01*dx[ind]);
270  }
271  size_t index = 0;
272  TIndex target = static_cast<TIndex>(m);
273  for(size_t j = 1; j < layer->children.size(); j++)
274  {
275  if(layer->children[j]->index == target)
276  {
277  index = j;
278  break;
279  }
280  }
281  //return std::make_pair(index, grids[ind][m] + (val01 - x) * (grids[ind][m + 1] - grids[ind][m]) / (y - x));
282  return std::make_pair(index, get_grid_value(ind, m) + (val01 - x) * (get_grid_value(ind, m + 1) - get_grid_value(ind, m)) / (y - x));
283  }

474  template <typename TIndex, typename TFloat>
475  std::pair<size_t, TFloat> ImplicitQuantileSorted<TIndex, TFloat>::quantile_transform(trie_based::NodeCount<TIndex> *layer, const std::vector<size_t> &psum, size_t ind, TFloat val01) const
476  {
477  size_t m = 0, count = grid_number[ind], step, c1 = 0, c2 = 0;
478  TFloat f1 = 0.0, f2 = 0.0, sample_size_u = static_cast<TFloat>(layer->count);
479  //auto first = grids[ind].begin();
480  //auto it = grids[ind].begin();
481  size_t it = 0, first = 0;
482 
483  while(count > 0)
484  {
485  it = first;
486  step = count / 2;
487  it += step;
488  m = it;
489 
490  //std::advance(it, step);
491  //m = std::distance(grids[ind].begin(), it);
492 
493  c1 = psum[count_less_binary(layer, m)];
494  f1 = c1/sample_size_u;
495 
496  if(f1 < val01)
497  {
498  c2 = psum[count_less_binary(layer, m + 1)];
499  f2 = c2/sample_size_u;
500 
501  if(val01 < f2)
502  break;
503 
504  first = ++it;
505  count -= step + 1;
506  }
507  else
508  {
509  count = step;
510  }
511  }
512 
513  if(count == 0)
514  {
515  c2 = psum[count_less_binary(layer, m + 1)];
516  f2 = c2/sample_size_u;
517  }
518 
519  if(c1 == c2)
520  {
521  if(c1 == 0)
522  {
523  return std::make_pair(0, get_grid_value(ind, layer->children.front()->index) + 2.0*val01*dx[ind]);
524  }
525  if(c1 == layer->count)
526  {
527  return std::make_pair(layer->children.size() - 1, get_grid_value(ind, layer->children.back()->index) + 2.0*val01*dx[ind]);
528  }
529 
530  TIndex target = static_cast<TIndex>(m);
531  auto pos = std::lower_bound(layer->children.begin(), layer->children.end(), target,
532  [](const std::shared_ptr<trie_based::NodeCount<TIndex>> &l,
533  const TIndex &r)
534  {
535  return l->index < r;
536  });
537  size_t index = std::distance(layer->children.begin(), pos);
538 
539  if(index > 0)
540  {
541  int curr1 = std::abs(static_cast<int>(layer->children[index]->index) - static_cast<int>(m));
542  int curr2 = std::abs(static_cast<int>(layer->children[index - 1]->index) - static_cast<int>(m));
543 
544  if(curr1 < curr2)
545  {
546  return std::make_pair(index, get_grid_value(ind, layer->children[index]->index) + 2.0*val01*dx[ind]);
547  }
548  else if(curr1 == curr2)
549  {
550  if(layer->children[index - 1]->index < layer->children[index]->index)
